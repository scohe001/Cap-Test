SELECT ROUND(SUM(t.Amount),2) [Actual Total], a.Total, a.Id, a.FirstName, a.LastName FROM Accounts a
JOIN Transactions t ON t.AccountId = a.Id
-- WHERE a.LastName = 'Prajzner'
GROUP BY t.AccountId --, t.Amount
HAVING ROUND(SUM(t.Amount),2) != a.Total
